  42. 42     }

  48. 48     {
  49. 49         $user = $token->getUser();
  50. 50 
  51. 51         // the user must be logged in; if not, deny permission
  52. 52         if (!$user instanceof User) {
  53. 53             return false;
  54. 54         }
  55. 55 
  56. 56         // the logic of this voter is pretty simple: if the logged user is the
